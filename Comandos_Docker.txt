//Construir Imagenes Docker
docker build -t cocacore7/backend_redes2_fase1 .
docker tag cocacore7/backend_redes2_fase1 cocacore7/backend_redes2_fase1
docker push cocacore7/backend_redes2_fase1
docker build -t cocacore7/frontend_redes2_fase1 .
docker tag cocacore7/frontend_redes2_fase1 cocacore7/frontend_redes2_fase1
docker push cocacore7/frontend_redes2_fase1

//Correr Imagenes Docker
sudo docker run --name backend_redes2_fase1 -p 4000:4000 -e "BD_HOST=ip_BD" -d cocacore7/backend_redes2_fase1 
sudo docker run -d -p 8000:80 --name frontend_redes2_fase1 cocacore7/frontend_redes2_fase1
